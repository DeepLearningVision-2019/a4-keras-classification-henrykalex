def stdrize(vector):
  mean = np.mean(vector)
  sigma = np.std(vector)
  for i in range(0, vector.size):
    vector[i] = (vector[i] - mean)/sigma
     
def clean_data(dataset):
  index = 0
  limit = dataset.shape[0]
  while(index < limit):
    flag = 0
    for j in range(0, dataset.shape[1]):
      if math.isnan(dataset[index][j]):
        flag = 1
    if flag:
      dataset = np.delete(dataset, index, axis=0)
      index = index - 1
      limit = limit - 1
    index = index + 1
  return dataset
